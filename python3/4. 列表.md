```python
"""
	列表常用方法
"""
#添加元素
ar = [1,2,3,4]
ar += [5]
# ar: [1,2,3,4,5]

ar.append(6)
#ar: [1,2,3,4,5,6]

ar.insert(7,7)
#ar: [1,2,3,4,5,6,7]

ar.extend([8,9])
#ar: [1,2,3,4,5,6,7,8,9]

ar[1]
# 2


#删除元素
ar.remove(9)
#ar: [1,2,3,4,5,6,7,8]

del ar[7]
#ar: [1,2,3,4,5,6,7]
>>> ar
[1, 2]
>>> del ar
>>> ar

  File "<stdin>", line 1, in <module>

>>> arr = [1,2,3]
>>> arr.pop()
3
>>> arr
[1, 2]

>>> arr
[1, 2, 3, 4, 5]

#分片操作
"""
	[s,e]从s索引值开始,到e索引值结束,不包括e索引值, 如果s,或者e没有初始, 那么默认为0和len(list)-1
"""
>>> arr = [1,2,3,4,5]
>>> arr
[1, 2, 3, 4, 5]
>>> arr[1:3]
[2, 3]
>>> arr[:4]
[1, 2, 3, 4]
>>> arr[1:]
[2, 3, 4, 5]



>>> ls = [1,2,3,4,5]
>>> ls.reverse()
>>> ls
[5, 4, 3, 2, 1]
>>> ls = [1,23,123,14,23,534,6,134,2]
>>> ls.sort()
>>> ls
[1, 2, 6, 14, 23, 23, 123, 134, 534]
>>> ls.sort(reverse=True)
>>> ls
[534, 134, 123, 23, 23, 14, 6, 2, 1]
>>> 
>>> 



```

##### 列表推导式

```python
#[有关 A 的表达式 for A in B] 
[i**i for i in range[2]]
# [0,1,4]

```

##### 浅拷贝和深拷贝

```python
>>> ls1 = ls #浅拷贝, 改变一个另一个改变
>>> ls1
[534, 134, 123, 23, 23, 14, 6, 2, 1]
>>> ls.sort()
>>> ls
[1, 2, 6, 14, 23, 23, 123, 134, 534]
>>> ls1
[1, 2, 6, 14, 23, 23, 123, 134, 534]
>>> 
>>> 
>>> ls1 = ls[:] # 深拷贝 或者ls.copy()
>>> ls.sort(reverse = True)
>>> ls
[534, 134, 123, 23, 23, 14, 6, 2, 1]
>>> ls1
[1, 2, 6, 14, 23, 23, 123, 134, 534]
```
