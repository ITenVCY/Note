##### setattr

```python
class  Re:
	def __init__(self,w = 0,h = 0):
		self.w = w
		self.h = h

	def __setattr__(self,name,value):
		print(str(name),str(value))
		if name == "sq":
			self.w = value
			self.h = value
		else:
			#self.name = value  #会陷入无限的递归中, 因为只要赋值操作就会调用__setattr__()
             #防止无限递归, self.__dict__[name] = value 
			super().__setattr__(name,value) #此处传入参数个数为2个
			
	def getS(self):
		return self.w * self.h


re = Re(4,5)
#re.sq = 2

print(re.getS())
```
