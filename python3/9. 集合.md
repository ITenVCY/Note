##### 序列去重

```python
#无集合时去重
def disNum(data):
	temp = 0
	dic = {}
	for it in data:
		dic[it] = temp
		temp += 1
	return list(dic.keys())

data = [1,213,12,312,312,423,1,23,1,34,12,412,4,34,12,31,123,1,21]
print(data)
data = disNum(data)
print(data)  

#无集合去重
def disNum(data):
	temp = []
	for it in data:
		if it not in temp:
        	temp.append(it)
	return temp

data = [1,213,12,312,312,423,1,23,1,34,12,412,4,34,12,31,123,1,21]
print(data)
data = disNum(data)
print(data)  

#输出
[1, 213, 12, 312, 312, 423, 1, 23, 1, 34, 12, 412, 4, 34, 12, 31, 123, 1, 21]
[1, 213, 12, 312, 423, 23, 34, 412, 4, 31, 123, 21]



#使用集合去重
def disNum(data):
	return list(set(data))

data = [1,213,12,312,312,423,1,23,1,34,12,412,4,34,12,31,123,1,21]
print(data)
data = disNum(data)
print(data) 
#输出
[1, 213, 12, 312, 312, 423, 1, 23, 1, 34, 12, 412, 4, 34, 12, 31, 123, 1, 21]
[1, 213, 12, 312, 423, 23, 34, 412, 4, 31, 123, 21]
```

##### 增加删除等

```python
>>> st = {1,2,312,3}
>>> st.add(4)
>>> st
{1, 2, 3, 4, 312}
>>> st.remove(1)
>>> st
{2, 3, 4, 312}

```
